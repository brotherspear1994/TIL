### 트리와 중위순회 만들기

```python
def is_empty():
    return lastidx == 0

def is_full(): # 트리가 가득찼는지 확인
    return lastidx == size

def add(n): # 원소 추가
    global lastidx
    if is_full():
        print("트리가 포화"); return
    lastidx += 1
    tree[lastidx] = n
## 전위 순회
def preorder(idx): # 부모,왼,오른쪽 => VLR
    if idx <= lastidx:
        # 현재노드 방문(출력)
        print(tree[idx], end =" ")
        #왼쪽
        preorder(2*idx)
        preorder(2*idx+1)
# 중위순회
def inorder(idx): # 왼,부,오 => LVR
    if idx <= lastidx:
        # 현재노드 방문(출력)
        #왼쪽
        inorder(2*idx)
        print(tree[idx], end=" ")
        inorder(2*idx+1)

# 후위순회
def postorder(idx):
    if idx <= lastidx:
        postorder(2*idx)
        postorder(2*idx+1)
        print(tree[idx], end=" ")

size = 15 #배열크기 지정
tree = [0]*(size+1)
lastidx = 0
# 원소 넣기: A,B,C,D...
for i in range(0,10):
    # 원소넣기
    # print(chr(i+65))
    add(chr(i+65))

print(tree)
preorder(1)
print()
inorder(1)
print()
postorder(1)
```

- 트리와 순회 알고리즘을 처음으로 학습하고 이를 직접 코드로 구현해 봤다.
- 전위순회,중위순회,후위순회의 알고리즘 작동과정을 직접 손으로 그려보며 어떻게 코드가 구현되는지 이해하려고 노력했다.

### 이진트리

```python
def preorder(node):
    if node:
        print(node,end=" ")
        preorder(tree[node][0])
        preorder(tree[node][1])

V = int(input())
E = V-1
tree = [[0]*3 for _ in range(V+1)]

temp = list(map(int,input().split()))
# tree에 저장
for i in range(E):
    p,c = temp[i*2], temp[i*2+1]
    if tree[p][0] == 0:
        tree[p][0] = c
    else:
        tree[p][1] = c
    tree[c][2] = p
for t in tree:
    print(*t)
preorder(1)
```

### heap

```python
def heappush(value):
    global heapcount
    heapcount += 1
    heap[heapcount] = value
    child = heapcount
    parent = child//2
    if parent and heap[parent] > heap[child]:
        while parent and heap[parent] > heap[child]:
            heap[parent], heap[child] = heap[child], heap[parent]
            child = parent
            parent = child//2
def heappop():
    global heapcount
    retValue = heap[1]
    heap[1] = heap[heapcount]
    heap[heapcount] = 0
    heapcount -= 1
    parent = 1
    child = parent*2
    if child+1 <= heapcount and heap[child] > heap[child+1]:
        child = child + 1
    if child <= heapcount and heap[parent]>heap[child]:
        while child <= heapcount and heap[parent]>heap[child]:
            heap[parent], heap[child] = heap[child], heap[parent]
            parent = child
            child = parent*2
            if child+1 <= heapcount and heap[child] > heap[child + 1]:
                child = child + 1
    return retValue



temp = [7,2,5,3,4,6]
N = len(temp)
heapcount = 0
heap = [0]*(N+1)
for i in range(N):
    heappush(temp[i])
print(heap)
for i in range(N):
    print(heappop(),end=" ")
print()
```

- heap 알고리즘(함수)를 직접 코드로 구현해 보았다.
- 어려운 개념에 코드길이도 길어 직접 구현하는데 까지 시간이 다소 걸렸다.

#### heapq 함수

```python
import heapq
heap1 = [7,2,5,3,4,6]
print(heap1)
heapq.heapify(heap1)
print(heap1)
heapq.heappush(heap1,1)
print(heap1)
while heap1:
    print(heapq.heappop(heap1), end=" ")
print()
##########################################
temp = [7,2,5,3,4,6]
heap2 = []
for i in range(len(temp)):
    heapq.heappush(heap2, (-temp[i],temp[i]))
heapq.heappush(heap2,(-1,1))
print(heap2)
while heap2:
    print(heapq.heappop(heap2)[1],end=" ")
```
